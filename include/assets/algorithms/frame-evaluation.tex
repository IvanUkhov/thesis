\begin{algorithm}[t]
  \caption{Evaluation of an adaptive hierarchical interpolant}
  \alab{frame-evaluation}
  \begin{pseudocode}
    \Require{interpolant, points}
    \Ensure{estimates} \Comment{approximated values}
    \For{point \textbf{in} points}
      \Let{estimate}{0}
      \For{(index, surplus) \textbf{in} interpolant}
        \Let{weight}{basis.Compute(index, point)}
        \Let{estimate}{$\text{estimate} + \text{surplus} \times \text{weight}$}
      \EndFor
      \State estimates.Append(estimate)
    \EndFor
    \State \Return estimates
  \end{pseudocode}
\end{algorithm}
