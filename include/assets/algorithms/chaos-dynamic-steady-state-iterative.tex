\begin{algorithm}[t]
  \caption{
    Calculation of a dynamic steady-state power or temperature profile for an
    outcome of the uncertain parameters.
  }
  \alab{chaos-dynamic-steady-state-iterative}
  \begin{algorithmic}[1]
    \vspace{0.4em}
    \Require{$\vu \in \real^\nu$}
    \Ensure{$\mq \in \real^{\np \times \ns}$ or $\mp \in \real^{\np \times \ns}$}
    \vspace{0.5em}
    \Let{$\mq$}{$\mq_\ambient$} \Comment{a matrix version of $\vq_\ambient$}
    \Repeat
      \Let{$\mp$}{$f(\vu, \mq)$} \Comment{for all time steps at once}
      \Let{$\mq$}{call \aref{dynamic-steady-state} for $\mp$}
    \Until{a stopping condition is satisfied}
    \vspace{0.4em}
  \end{algorithmic}
\end{algorithm}
