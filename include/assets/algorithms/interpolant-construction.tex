\begin{algorithm}[t]
  \caption{Construction of a hierarchical interpolant.}
  \alab{interpolant-construction}
  \begin{algorithmic}[1]
    \Require{Algorithm~G} \Comment{a subroutine evaluating $\g \circ \transform$}
    \Ensure{interpolant}
    \While{strategy.Continue?(state)}
      \Let{state}{strategy.Next(state)}
      \Let{state.nodes}{grid.Compute(state.indices)}
      \Let{state.values}{Algorithm~G(state.nodes)}
      \Let{state.estimates}{Evaluate(interpolant, state.nodes)}
      \Let{state.surpluses}{Subtract(state.values, state.estimates)}
      \Let{state.scores}{strategy.Score(state.surpluses)}
      \State interpolant.Append(state.indices, state.surpluses)
    \EndWhile
    \State \Return interpolant
  \end{algorithmic}
\end{algorithm}
