\begin{algorithm}[t]
  \caption{Construction of a hierarchical interpolant of a function.}
  \alab{interpolant-construction}
  \begin{algorithmic}[1]
    \Require{target} \Comment{a function to interpolate}
    \Ensure{surrogate} \Comment{an interpolant}
    \Let{surrogate}{New()}
    \For{s $\gets$ strategy.First(); s $\neq$ nil; s $\gets$ strategy.Next(s)}
      \Let{s.nodes}{grid.Compute(s.indices)}
      \Let{s.values}{Invoke(target, s.nodes)}
      \Let{s.estimates}{Evaluate(surrogate, s.nodes)}
      \Let{s.surpluses}{Subtract(s.values, s.estimates)}
      \Let{s.scores}{strategy.Score(s.surpluses)}
      \State Append(surrogate, s.indices, s.surpluses)
    \EndFor
    \State return surrogate
  \end{algorithmic}
\end{algorithm}
