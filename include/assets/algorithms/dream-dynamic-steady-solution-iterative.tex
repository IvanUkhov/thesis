\begin{algorithm}[t]
  \caption{
    Calculation of dynamic steady-state power and temperature profiles
    considering the power-temperature interdependence
  }
  \alab{dream-dynamic-steady-solution-iterative}
  \begin{pseudocode}
    \Require{$\mp_\dynamic \in \real^{\np \times \ns}$}
    \Ensure{$\mp \in \real^{\np \times \ns}$, $\mq \in \real^{\np \times \ns}$}
    \Let{$\mq$}{$\mq_\ambient$} \Comment{a matrix version of $\vq_\ambient$}
    \Repeat
      \Let{$\mp$}{$\mp_\dynamic + \mp_\static(\mq)$}
      \Let{$\mq$}{\aref{dream-dynamic-steady-solution}($\mp$)}
    \Until{a stopping condition is satisfied}
    \State \Return $\mp$, $\mq$
  \end{pseudocode}
\end{algorithm}
