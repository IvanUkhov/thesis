\begin{algorithm}[t]
  \caption{Evaluation of a hierarchical interpolant at a set of points.}
  \alab{interpolant-evaluation}
  \begin{algorithmic}[1]
    \Require{surrogate, points} \Comment{an interpolant and points}
    \Ensure{estimates} \Comment{approximated values}
    \Let{estimates}{New()}
    \For{point in points}
      \Let{estimate}{0}
      \For{(index, surplus) in surrogate}
        \Let{weight}{basis.Compute(index, point)}
        \Let{estimate}{$\text{estimate} + \text{surplus} \times \text{weight}$}
      \EndFor
      \State Append(estimates, estimate)
    \EndFor
    \State return estimates
  \end{algorithmic}
\end{algorithm}
