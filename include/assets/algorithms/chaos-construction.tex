\begin{algorithm}[t]
  \caption{Construction of a polynomial chaos expansion.}
  \alab{chaos-construction}
  \begin{algorithmic}[1]
    \Require{Algorithm~G} \Comment{a subroutine evaluating $\g \circ \transform$}
    \Ensure{$\hat{\vg} \in \real^\nc$}
    \For{$i \gets 1 \text{ to } \nq$} \Comment{for each quadrature point $\vx_i$}
      \Let{$\vg(i)$}{Algorithm~G($\vx_i$)}
    \EndFor
    \Let{$\hat{\vg}$}{$\Pi \vg$}
    \State \Return $\hat{\vg}$
  \end{algorithmic}
\end{algorithm}
