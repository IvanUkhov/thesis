\begin{algorithm}[t]
  \caption{Construction of a polynomial chaos expansion}
  \alab{chaos-construction}
  \begin{pseudocode}
    \Require{Algorithm~G} \Comment{a subroutine evaluating $\g \circ \transform$}
    \Ensure{$\hat{\vg} \in \real^\nc$}
    \For{$i \gets 1 \textbf{ to } \nq$} \Comment{for each quadrature point $\vz_i$}
      \Let{$\vg(i)$}{Algorithm~G($\vz_i$)}
    \EndFor
    \Let{$\hat{\vg}$}{$\Pi \vg$}
    \State \Return $\hat{\vg}$
  \end{pseudocode}
\end{algorithm}
