\begin{algorithm}[t]
  \caption{
    Calculation of a dynamic steady-state power and temperature profile for an
    outcome of the uncertain parameters
  }
  \alab{chaos-dynamic-steady-solution-iterative}
  \begin{algorithmic}[1]
    \Require{$\vu \in \real^\nu$}
    \Ensure{$\mp \in \real^{\np \times \ns}$, $\mq \in \real^{\np \times \ns}$}
    \Let{$\mq$}{$\mq_\ambient$} \Comment{a matrix version of $\vq_\ambient$}
    \Repeat
      \Let{$\mp$}{$\f(\vu, \mq)$} \Comment{for all time steps at once}
      \Let{$\mq$}{\aref{dream-dynamic-steady-solution}($\mp$)}
    \Until{a stopping condition is satisfied}
    \State \Return $\mp$, $\mq$
  \end{algorithmic}
\end{algorithm}
